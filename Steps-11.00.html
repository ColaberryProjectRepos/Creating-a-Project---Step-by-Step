Here is the code on how to combine 2 files 
import pandas as pd

# Read the CSV files into DataFrames
df1 = pd.read_csv('/lakehouse/default/Files/DataWithIndex_Final.csv/part-00000-c42d20aa-3d3f-4172-ac75-37389fbeed40-c000.csv')
df2 = pd.read_csv('/lakehouse/default/Files/DataWithIndex_Walmart.csv/part-00000-82392eba-2e6e-42ff-a498-8cf8907f496d-c000.csv')

# Combine the DataFrames based on their indices
combined_df = df1.merge(df2, left_index=True, right_index=True)

# Save the combined DataFrame to a new CSV file
combined_df.to_csv('/lakehouse/default/Files/Final_walmart_test.csv', index=False)
